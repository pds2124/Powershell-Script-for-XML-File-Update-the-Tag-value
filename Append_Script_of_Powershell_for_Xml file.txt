param(
[string]$ClientCode,
[string]$San,
[string]$XMLFile
)
[xml]$xml =Get-Content $XMLFile
$b =   $San + "_" + $ClientCode
# this code is used to append "SAN" before the "Gly" in "XMLFile"
foreach($item in (Select-XML -Xml $xml -XPath '//object-stream'))
{
    if(($item -match $ClientCode) -and ($item -notmatch $San))
    {
    $word = $ClientCode
    $replacement = $b
    $item.Node.InnerXml = $item -replace $word,$replacement
    } 
     
}
foreach($item in (Select-XML -Xml $xml -XPath '//Except_Tag'))
{
    if($item -match $ClientCode) 
    {
    $word = $b
    $replacement = $ClientCode
    $item.Node.InnerText = $item -replace $word,$replacement
    }  
}
$xml.Save($XMLFile);
